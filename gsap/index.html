CTA and mouseover event

function fl_ClickToGoToWebPage() {
   window.open(window.clickTag);
}
function mOver() {
   exportRoot.mc.gotoAndPlay(1);
}
this.cta.addEventListener("click", fl_ClickToGoToWebPage);
this.cta.addEventListener("mouseover", mOver);
window.addEventListener("mouseover", mOver);
